parsec-inspired elisp pkg

* Installation
#+begin_src emacs-lisp
  (add-to-list 'load-path "/path/to/lp")
  (require 'lp)
#+end_src

* Usage
#+name: parse-k
#+begin_src emacs-lisp
  (defun parse-k (k)
    "parse some k"
        (lp-with-input k
          (let ((f "+-*%&|<>=~!,@?^#_$")
                (g '("'/\\")))
            (lp-or (lp-n)
             (lp-c-is-in f)
             (lp-c-is-in g)))))
#+end_src
