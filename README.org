#+TITLE: lang parser
#+DATE: 2023-01-31
parsec-inspired elisp pkg

* Installation
#+begin_src emacs-lisp
  (add-to-list 'load-path "/path/to/lp")
  (require 'lp)
#+end_src

* Usage
#+name: parse-k
#+begin_src emacs-lisp
  (defun parse-k (k)
    "parse some k"
        (lp-with-input k
          (let ((f (lp-s2l "+-*%&|<>=~!,@?^#_$"))
                (g (lp-s2l "'/\\"))
                (com ";.*"))
            (lp-or (lp-c-l f)
                   (lp-c-l g)
                   (lp-re com)
                   (lp-n)))))
#+end_src
